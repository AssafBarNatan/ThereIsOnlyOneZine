\section{Calculus}

\begin{thm}
  Every sequence converges.
\end{thm}
\begin{proof}
  Consider a sequence: $1,1,\ldots$. Let $\eps=1$ be arbitrary, 
  we have:
  \begin{align*}
    |1-1| = |1| = 1 = \eps
  \end{align*}
\end{proof}
\vspace{-0.5cm}
